# robots.txt for Wikantik
# Allow indexing of public wiki content while blocking administrative/action pages

User-agent: *

# Allow the sitemap
Sitemap: https://wiki.wikantik.com/sitemap.xml

# Block administrative pages
Disallow: /admin/

# Block editing, diff, and action paths (not content)
Disallow: /edit/
Disallow: /diff/
Disallow: /preferences
Disallow: /reset-password

# Block API and internal endpoints.
# EXCEPT the read-only page-content API: the React reader fetches
# /api/pages/{name}?render=true to render page bodies. Google's renderer (WRS)
# skips robots-disallowed resources, so blocking this made rendered pages look
# empty and Google flagged them as "Soft 404". Keep it crawlable; the Allow must
# precede the Disallow so first-match crawlers honour it.
Allow: /api/pages/
Disallow: /api/
Disallow: /mcp
Disallow: /metrics

# Block search results pages (let Google use its own index)
Disallow: /search

# Crawl-delay: be polite to crawlers (1 second between requests)
Crawl-delay: 1
